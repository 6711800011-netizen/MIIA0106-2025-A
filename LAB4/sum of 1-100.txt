#include <iostream>
using namespace std;

int main() {
    int sum = 0; // 1. สร้างตัวแปรเก็บผลรวม (ต้องเริ่มที่ 0 เสมอ)

    // 2. ใช้ for loop วนตั้งเเต่ 1 ถึง 100
    for (int i = 1; i <= 100; i++) {
        sum = sum + i; // 3. เอาเลขในรอบนั้นๆ (i) มาบวกสะสมเข้าไปใน sum
    }

    // 4. แสดงผลลัพธ์หลังจากวนลูปเสร็จแล้ว
    cout << "Sum of 1 to 100 is: " << sum << endl;

    return 0;
}
