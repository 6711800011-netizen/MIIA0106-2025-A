#include <iostream>
using namespace std;

int main() {
    int n = 11; // กำหนดให้เป็นแม่ 11

    cout << "Multiplication Table of 11" << endl;
    cout << "--------------------------" << endl;

    // วนลูปตั้งแต่ 1 ถึง 12
    for (int i = 1; i <= 12; i++) {
        // แสดงผลในรูปแบบ: 11 x i = ผลลัพธ์
        cout << n << " x " << i << " = " << (n * i) << endl;
    }

    return 0;
}
